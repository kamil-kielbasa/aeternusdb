1//! SSTable iterators — block-level and multi-block scan.
2//!
3//! This module provides two iterator types:
4//!
5//! - [`BlockIterator`] — decodes a single data block and yields [`BlockEntry`]
6//!   values. It supports `seek_to_first()` and `seek_to(key)` for positioning.
7//! - [`ScanIterator`] — walks multiple data blocks plus range tombstones,
8//!   yielding a merged stream of [`Record`] items in `(key ASC, LSN DESC)` order.
9//!
10//! # Block Iterator
11//!
12//! The block iterator operates on the raw bytes of a single data block.
13//! Each entry is an encoded `SSTableCell` header followed by key and value bytes.
14//!
15//! ```text
16//! [SSTableCell header][KEY_BYTES][VALUE_BYTES]
17//! ```
18//!
19//! The header contains fixed-integer-encoded metadata:
20//!
21//! - `key_len` (u32)
22//! - `value_len` (u32)
23//! - `lsn` (u64)
24//! - `timestamp` (u64)
25//! - `is_delete` (bool)
26//!
27//! Seeking is linear within a block. Blocks are intentionally small (typically
28//! 4 KiB), so linear search is efficient. If corruption or truncation is
29//! detected, the iterator treats the block as exhausted.
30//!
31//! # Scan Iterator
32//!
33//! [`ScanIterator`] provides a **sorted forward scan** over a single SSTable,
34//! yielding all point entries (`Put`, `Delete`) and range tombstones
35//! (`RangeDelete`) that overlap a user-specified key range `[start_key, end_key)`.
36//!
37//! Blocks are decoded lazily and sequentially. When a block is exhausted, the
38//! iterator advances to the next one automatically. Range tombstones are
39//! interleaved with point entries in key order.
40//!
41//! The scan iterator does **not** perform visibility resolution — that is the
42//! responsibility of upper layers (engine merge iterator, visibility filter).
43

54/// A fully decoded entry from a data block.
55///
56/// This is the in-memory representation of a single SSTable cell after decoding.
57#[derive(Debug, Clone, PartialEq, Eq)]
58pub struct BlockEntry {
59    /// The user key bytes.
60    pub key: Vec<u8>,
61
62    /// The value bytes. Empty for tombstones.
63    pub value: Vec<u8>,
64
65    /// Whether this entry represents a point delete.
66    pub is_delete: bool,
67
68    /// Log sequence number associated with this version.
69    pub lsn: u64,
70
71    /// Commit timestamp supplied by the storage engine.
72    pub timestamp: u64,
73}

75// ------------------------------------------------------------------------------------------------
76// Block Iterator
77// ------------------------------------------------------------------------------------------------
78
79/// Iterator over the entries contained within a single SSTable data block.
80///
81/// This iterator:
82///
83/// - Decodes `SSTableCell` boundaries using custom encoding with fixed-int encoding.
84/// - Provides block-local forward iteration.
85/// - Supports basic key seeking within the block.
86///
87/// It **does not** handle merging multiple blocks, range tombstones, bloom filter lookups,
88/// or other higher-level SSTable mechanics—those are implemented in the outer SSTable layer.
89pub struct BlockIterator {
90    /// Raw, decompressed block payload (entries only).
91    data: Vec<u8>,
92
93    /// Cursor into `data`, always pointing at the next header to decode.
94    cursor: usize,
95}
96
97impl BlockIterator {
98    /// Create a new iterator from already-decoded block bytes.
99    ///
100    /// The provided `data` slice must contain a concatenation of encoded `SSTableCell`s.
101    pub fn new(data: Vec<u8>) -> Self {
102        Self { data, cursor: 0 }
103    }
104
105    /// Reset the iterator to the first entry in the block.
106    pub fn seek_to_first(&mut self) {
107        self.cursor = 0;
108    }
109
110    /// Seek to the first entry whose key is **≥ `search_key`**.
111    ///
112    /// This performs a **linear scan**. If corruption or truncation is detected,
113    /// the iterator stops at the end of the block.
114    pub fn seek_to(&mut self, search_key: &[u8]) {
115        self.cursor = 0;
116        while self.cursor < self.data.len() {
117            match encoding::decode_from_slice::<SSTableCell>(&self.data[self.cursor..]) {
118                Ok((cell, cell_len)) => {
119                    let pos = self.cursor + cell_len;
120
121                    let key_len = cell.key_len as usize;
122                    let value_len = cell.value_len as usize;
123
124                    if pos + key_len + value_len > self.data.len() {
125                        // truncated -> treat as end
126                        self.cursor = self.data.len();
127                        return;
128                    }
129
130                    let key_bytes = &self.data[pos..pos + key_len];
131                    if key_bytes >= search_key {
132                        // leave cursor at start of this cell
133                        return;
134                    }
135
136                    // advance to next cell
137                    self.cursor = pos + key_len + value_len;
138                }
139                Err(e) => {
140                    tracing::error!(cursor = self.cursor, ?e, "decode error during seek");
141                    self.cursor = self.data.len();
142                    return;
143                }
144            }
145        }
146    }
147
148    /// Decode and return the next entry, advancing the cursor.
149    ///
150    /// Returns `None` if:
151    /// - the cursor is at or past the end of the block,
152    /// - decoding fails,
153    /// - the block appears truncated.
154    pub fn next_entry(&mut self) -> Option<BlockEntry> {
155        if self.cursor >= self.data.len() {
156            return None;
157        }
158
159        match encoding::decode_from_slice::<SSTableCell>(&self.data[self.cursor..]) {
160            Ok((cell, cell_len)) => {
161                self.cursor += cell_len;
162
163                let key_len = cell.key_len as usize;
164                let value_len = cell.value_len as usize;
165
166                if self.cursor + key_len + value_len > self.data.len() {
167                    // truncated -> treat as end
168                    self.cursor = self.data.len();
169                    return None;
170                }
171
172                let key = self.data[self.cursor..self.cursor + key_len].to_vec();
173                self.cursor += key_len;
174                let value = self.data[self.cursor..self.cursor + value_len].to_vec();
175                self.cursor += value_len;
176
177                Some(BlockEntry {
178                    key,
179                    value,
180                    is_delete: cell.is_delete,
181                    lsn: cell.lsn,
182                    timestamp: cell.timestamp,
183                })
184            }
185            Err(_) => {
186                // invalid encoding -> treat as end
187                self.cursor = self.data.len();
188                None
189            }
190        }
191    }
192
193    /// Returns `true` if the iterator has reached the end of the block or encountered corruption.
194    pub fn is_end(&self) -> bool {
195        self.cursor >= self.data.len()
196    }
197}
198
199/// Implements idiomatic Rust iteration over block entries.
200impl Iterator for BlockIterator {
201    type Item = BlockEntry;
202
203    fn next(&mut self) -> Option<Self::Item> {
204        self.next_entry()
205    }
206}

212/// Iterator over all SSTable entries (point or range tombstones)
213/// within the half-open interval:
214///
215/// ```text
216/// [start_key, end_key)
217/// ```
218///
219/// This iterator yields items of type [`Record`].
220///
221/// Internally, it:
222///
223/// - Tracks the current data-block index (`current_block_index`)
224/// - Holds a block-local iterator (`BlockIterator`)
225/// - Iterates through range tombstones stored in a separate structure
226///
227/// Errors during block loading or decoding are returned via the iterator.
228pub struct ScanIterator<'a> {
229    /// Reference to the SSTable being scanned.
230    sstable: &'a SSTable,
231
232    /// Current index into the SSTable block index.
233    current_block_index: usize,
234
235    /// Iterator over the entries in the current data block.
236    current_block_iter: Option<BlockIterator>,
237
238    /// Left bound of the user scan (inclusive).
239    start_key: Vec<u8>,
240
241    /// Right bound of the user scan (exclusive).
242    end_key: Vec<u8>,
243
244    /// Index into the SSTable range tombstone array.
245    pending_range_idx: usize,
246
247    /// Next range tombstone to yield.
248    next_range: Option<Record>,
249
250    /// Next point entry (Put/Delete) to yield.
251    next_point: Option<Record>,
252}
253
254impl<'a> ScanIterator<'a> {
255    /// Create a new SSTable scan iterator for the half-open range
256    /// `start_key <= key < end_key`.
257    pub fn new(
258        sstable: &'a SSTable,
259        start_key: Vec<u8>,
260        end_key: Vec<u8>,
261    ) -> Result<Self, SSTableError> {
262        if start_key >= end_key {
263            return Err(SSTableError::Internal("scan start >= end".to_string()));
264        }
265
266        let current_block_index = sstable.find_block_for_key(start_key.as_slice());
267
268        let block_iter = if current_block_index < sstable.index.len() {
269            let entry = &sstable.index[current_block_index];
270            let block_bytes = SSTable::read_block_bytes(&sstable.mmap, &entry.handle)?;
271            let (block, _) = encoding::decode_from_slice::<SSTableDataBlock>(&block_bytes)?;
272            let mut it = BlockIterator::new(block.data);
273            it.seek_to(start_key.as_slice());
274            Some(it)
275        } else {
276            None
277        };
278
279        Ok(Self {
280            sstable,
281            current_block_index,
282            current_block_iter: block_iter,
283            start_key,
284            end_key,
285            pending_range_idx: 0,
286            next_range: None,
287            next_point: None,
288        })
289    }
290
291    /// Load the next data block and create a fresh `BlockIterator`.
292    fn load_next_block(&mut self) -> Result<bool, SSTableError> {
293        self.current_block_index += 1;
294
295        if self.current_block_index >= self.sstable.index.len() {
296            self.current_block_iter = None;
297            return Ok(false);
298        }
299
300        let entry = &self.sstable.index[self.current_block_index];
301        let block_bytes = SSTable::read_block_bytes(&self.sstable.mmap, &entry.handle)?;
302
303        let (block, _) = encoding::decode_from_slice::<SSTableDataBlock>(&block_bytes)?;
304        let mut it = BlockIterator::new(block.data);
305        it.seek_to_first();
306        self.current_block_iter = Some(it);
307
308        Ok(true)
309    }
310
311    /// Return the next *point entry* (Put/Delete) in the scan key range,
312    /// automatically advancing to the next block as needed.
313    fn next_point_or_delete(&mut self) -> Option<Record> {
314        loop {
315            let it = self.current_block_iter.as_mut()?;
316
317            if let Some(item) = it.next_entry() {
318                // Stop when out of scan range
319                if item.key.as_slice() >= self.end_key.as_slice() {
320                    return None;
321                }
322
323                if item.is_delete {
324                    return Some(Record::Delete {
325                        key: item.key,
326                        lsn: item.lsn,
327                        timestamp: item.timestamp,
328                    });
329                }
330
331                return Some(Record::Put {
332                    key: item.key,
333                    value: item.value,
334                    lsn: item.lsn,
335                    timestamp: item.timestamp,
336                });
337            }
338
339            // end of block - load next
340            match self.load_next_block() {
341                Ok(true) => {}
342                Ok(false) | Err(_) => return None,
343            }
344        }
345    }
346
347    /// Return the next range tombstone that overlaps the scan range.
348    fn next_range_delete(&mut self) -> Option<Record> {
349        while self.pending_range_idx < self.sstable.range_deletes.data.len() {
350            let r = &self.sstable.range_deletes.data[self.pending_range_idx];
351
352            // Skip ranges completely left of scan window
353            if r.end_key.as_slice() <= self.start_key.as_slice() {
354                self.pending_range_idx += 1;
355                continue;
356            }
357
358            // Stop when range start is beyond end of scan range
359            if r.start_key.as_slice() >= self.end_key.as_slice() {
360                return None;
361            }
362
363            // Emit range
364            self.pending_range_idx += 1;
365
366            return Some(Record::RangeDelete {
367                start: r.start_key.clone(),
368                end: r.end_key.clone(),
369                lsn: r.lsn,
370                timestamp: r.timestamp,
371            });
372        }
373
374        None
375    }
376
377    /// Ensure that `next_range` is populated.
378    fn fill_range(&mut self) {
379        if self.next_range.is_none() {
380            self.next_range = self.next_range_delete();
381        }
382    }
383
384    /// Ensure that `next_point` is populated.
385    fn fill_point(&mut self) {
386        if self.next_point.is_none() {
387            self.next_point = self.next_point_or_delete();
388        }
389    }
390}
391
392impl<'a> Iterator for ScanIterator<'a> {
393    type Item = Record;
394
395    fn next(&mut self) -> Option<Self::Item> {
396        self.fill_range();
397        self.fill_point();
398
399        match (&self.next_range, &self.next_point) {
400            (None, None) => None, // end of scan
401
402            (Some(_), None) => self.next_range.take(),
403            (None, Some(_)) => self.next_point.take(),
404
405            (Some(r), Some(p)) => {
406                if r.key()
407                    .cmp(p.key())
408                    .then_with(|| p.lsn().cmp(&r.lsn()))
409                    .is_le()
410                {
411                    self.next_range.take()
412                } else {
413                    self.next_point.take()
414                }
415            }
416        }
417    }
