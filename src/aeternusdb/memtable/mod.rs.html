1//! # Memtable Module
2//!
3//! ## Design Invariants
4//!
5//! - All writes are WAL-first and assigned a monotonically increasing LSN.
6//! - The memtable may contain multiple versions per key; the highest-LSN
7//!   version is considered authoritative.
8//! - Deletes are represented via tombstones, not physical removal.
9//! - Range tombstones logically delete all keys in `[start, end)`
10//!   with lower LSNs.
11//! - Reads (`get`, `scan`) always resolve point entries against
12//!   range tombstones.
13//!
14//! ## Flush Semantics
15//!
16//! - `iter_for_flush` returns a *logical snapshot* of the memtable state.
17//! - Returned records are sufficient to reconstruct the same memtable
18//!   state via WAL replay.
19//! - Flush iteration does **not** mutate or clear in-memory state.
20//!
21//! ## Frozen Memtable
22//!
23//! - A `FrozenMemtable` is read-only.
24//! - It retains ownership of the WAL to guarantee durability until
25//!   data is persisted to SSTables.
26

58/// Represents possible errors returned by [`Memtable`] operations.
59#[derive(Debug, Error)]
60pub enum MemtableError {
61    /// Underlying WAL I/O failure.
62    #[error("WAL error: {0}")]
63    Wal(#[from] WalError),
64
65    /// Write buffer limit reached; a flush is required before further writes.
66    #[error("Flush required")]
67    FlushRequired,
68
69    /// Internal invariant violation or poisoned lock.
70    #[error("Internal error: {0}")]

78/// The mutable, in-memory write buffer of the storage engine.
79///
80/// The memtable:
81/// - Accepts writes (`put`, `delete`, `delete_range`)
82/// - Persists all mutations to a WAL
83/// - Serves reads (`get`, `scan`)
84/// - Can be logically flushed via `iter_for_flush`
85///
86/// Internally, the memtable stores **multiple versions per key** ordered
87/// by descending LSN. Resolution is deferred to read time.
88///
89/// # Concurrency
90/// - Writers acquire an exclusive lock
91/// - Readers may proceed concurrently
92///
93/// # Durability
94/// - Every mutation is appended to the WAL *before* being applied in memory
95pub struct Memtable {
96    /// Thread-safe container for in-memory data and metadata.
97    inner: Arc<RwLock<MemtableInner>>,
98
99    /// Associated write-ahead log for durability.
100    pub wal: Wal<Record>,
101
102    /// Monotonic log sequence number (LSN) for version ordering.
103    next_lsn: AtomicU64,
104}
105
106/// A single versioned point entry stored in the memtable.
107///
108/// A key may have multiple `MemtableSingleEntry` versions, ordered by LSN.
109/// The highest-LSN entry is considered the latest.
110///
111/// Deletions are represented by tombstones (`is_delete = true`).
112#[derive(Debug, PartialEq, Clone)]
113pub struct MemtableSingleEntry {
114    /// The stored value. `None` indicates a deletion (tombstone).
115    pub value: Option<Vec<u8>>,
116
117    /// Logical timestamp in nanoseconds since UNIX epoch.
118    pub timestamp: u64,
119
120    /// Whether this entry represents a deletion.
121    pub is_delete: bool,
122
123    /// Log sequence number for ordering updates.
124    pub lsn: u64,
125}

137impl crate::encoding::Decode for MemtableSingleEntry {
138    fn decode_from(buf: &[u8]) -> Result<(Self, usize), crate::encoding::EncodingError> {
139        let (value, mut offset) = <Option<Vec<u8>> as crate::encoding::Decode>::decode_from(buf)?;
140        let (timestamp, n) = <u64 as crate::encoding::Decode>::decode_from(&buf[offset..])?;
141        offset += n;
142        let (is_delete, n) = <bool as crate::encoding::Decode>::decode_from(&buf[offset..])?;
143        offset += n;
144        let (lsn, n) = <u64 as crate::encoding::Decode>::decode_from(&buf[offset..])?;
145        offset += n;
146        Ok((
147            MemtableSingleEntry {
148                value,
149                timestamp,
150                is_delete,
151                lsn,
152            },
153            offset,
154        ))
155    }

160/// Result of a `get` operation on the memtable.
161#[derive(Debug, PartialEq)]
162pub enum MemtableGetResult {
163    /// Value found for the key.
164    Put(Vec<u8>),
165
166    /// Key was deleted by a point tombstone.

168
169    /// Key was deleted by a range tombstone.
170    RangeDelete,
171
172    /// Key not found in the memtable.
173    NotFound,

176/// Internal shared state of the memtable.
177///
178/// This structure is protected by an `RwLock` and must never be
179/// accessed directly outside the memtable implementation.
180struct MemtableInner {
181    /// Point entries grouped by key, then ordered by descending LSN.
182    tree: BTreeMap<Vec<u8>, BTreeMap<Reverse<u64>, MemtableSingleEntry>>,
183
184    /// Range tombstones indexed by start key and ordered by descending LSN.
185    range_tombstones: BTreeMap<Vec<u8>, BTreeMap<Reverse<u64>, RangeTombstone>>,
186
187    /// Approximate in-memory footprint.
188    approximate_size: usize,
189
190    /// Configured maximum buffer size before flush is required.
191    write_buffer_size: usize,
192}

194impl Memtable {
195    /// Creates a new mutable [`Memtable`] backed by a write-ahead log (WAL).
196    ///
197    /// # Arguments
198    /// - `wal_path` — Path to the WAL file used for durability.
199    /// - `max_record_size` — Optional maximum size of a single WAL record.
200    /// - `write_buffer_size` — Maximum in-memory size before a flush is required.
201    ///
202    /// # Behavior
203    /// - Replays the WAL (if present) to reconstruct the in-memory state.
204    /// - Restores the highest observed LSN and advances the internal counter.
205    /// - Subsequent writes will continue with monotonically increasing LSNs.
206    ///
207    /// # Crash Safety
208    /// WAL replay guarantees recovery to the last durable state after a crash.
209    pub fn new<P: AsRef<Path>>(
210        wal_path: P,
211        max_record_size: Option<u32>,
212        write_buffer_size: usize,
213    ) -> Result<Self, MemtableError> {
214        info!("Initializing Memtable with WAL replay");
215
216        let wal = Wal::open(&wal_path, max_record_size)?;
217
218        let mut inner = MemtableInner {
219            tree: BTreeMap::new(),
220            range_tombstones: BTreeMap::new(),
221            approximate_size: 0,
222            write_buffer_size,
223        };
224
225        let mut max_lsn_seen: u64 = 0;
226
227        let records = wal.replay_iter()?;
228        for record in records {
229            let record: Record = record?;
230
231            match record {
232                Record::Put {
233                    key,
234                    value,
235                    lsn,
236                    timestamp,
237                } => {
238                    let record_size =
239                        std::mem::size_of::<MemtableSingleEntry>() + key.len() + value.len();
240                    inner.approximate_size += record_size;
241
242                    if lsn > max_lsn_seen {
243                        max_lsn_seen = lsn;
244                    }
245
246                    let record_value = MemtableSingleEntry {
247                        value: Some(value),
248                        timestamp,
249                        is_delete: false,
250                        lsn,
251                    };
252
253                    inner
254                        .tree
255                        .entry(key)
256                        .or_default()
257                        .insert(Reverse(record_value.lsn), record_value);
258                }
259
260                Record::Delete {
261                    key,
262                    lsn,
263                    timestamp,
264                } => {
265                    let record_size = std::mem::size_of::<MemtableSingleEntry>() + key.len();
266                    inner.approximate_size += record_size;
267
268                    if lsn > max_lsn_seen {
269                        max_lsn_seen = lsn;
270                    }
271
272                    let record_value = MemtableSingleEntry {
273                        value: None,
274                        timestamp,
275                        is_delete: true,
276                        lsn,
277                    };
278
279                    inner
280                        .tree
281                        .entry(key)
282                        .or_default()
283                        .insert(Reverse(record_value.lsn), record_value);
284                }
285
286                Record::RangeDelete {
287                    start,
288                    end,
289                    lsn,
290                    timestamp,
291                } => {
292                    let record_size =
293                        std::mem::size_of::<RangeTombstone>() + start.len() + end.len();
294                    inner.approximate_size += record_size;
295
296                    if lsn > max_lsn_seen {
297                        max_lsn_seen = lsn;
298                    }
299
300                    let record_value = RangeTombstone {
301                        start,
302                        end,
303                        lsn,
304                        timestamp,
305                    };
306
307                    inner
308                        .range_tombstones
309                        .entry(record_value.start.clone())
310                        .or_default()
311                        .insert(Reverse(record_value.lsn), record_value);
312                }
313            }
314        }
315
316        info!(
317            "Memtable initialized successfully with LSN: {}",
318            max_lsn_seen
319        );
320
321        Ok(Self {
322            inner: Arc::new(RwLock::new(inner)),
323            wal,
324            next_lsn: AtomicU64::new(max_lsn_seen.saturating_add(1)),
325        })
326    }

328    /// Inserts or updates a key with a new value.
329    ///
330    /// # Behavior
331    /// - The mutation is first appended to the WAL (write-ahead).
332    /// - The entry is then applied to the in-memory balanced tree.
333    /// - A unique, monotonically increasing LSN is assigned.
334    pub fn put(&self, key: Vec<u8>, value: Vec<u8>) -> Result<(), MemtableError> {
335        trace!("put() started, key: {}", HexKey(&key));
336
337        if key.is_empty() || value.is_empty() {
338            return Err(MemtableError::Internal("Key or value is empty".to_string()));
339        }
340
341        let lsn = self.next_lsn.fetch_add(1, Ordering::SeqCst);
342        let timestamp = Self::current_timestamp();
343
344        let record_size = std::mem::size_of::<MemtableSingleEntry>() + key.len() + value.len();
345        let record = Record::Put {
346            key: key.clone(),
347            value: value.clone(),
348            timestamp,
349            lsn,
350        };
351
352        let mut guard = self.inner.write().map_err(|_| {
353            error!("Read-write lock poisoned during put");
354            MemtableError::Internal("Read-write lock poisoned".into())
355        })?;
356
357        if guard.approximate_size + record_size > guard.write_buffer_size {
358            return Err(MemtableError::FlushRequired);
359        }
360
361        // 1. Wal first (crash safety)
362        self.wal.append(&record)?;
363
364        // 2. In-memory update
365        let value = MemtableSingleEntry {
366            value: Some(value),
367            timestamp,
368            is_delete: false,
369            lsn,
370        };
371
372        guard
373            .tree
374            .entry(key.clone())
375            .or_default()
376            .insert(Reverse(value.lsn), value);
377
378        guard.approximate_size += record_size;
379
380        trace!(
381            "Put operation completed with LSN: {}, key: {}",
382            lsn,
383            HexKey(&key)
384        );
385
386        Ok(())
387    }

449    /// Deletes all keys in the range `[start, end)`.
450    ///
451    /// # Range Semantics
452    /// - Inclusive `start`
453    /// - Exclusive `end`
454    ///
455    /// # Behavior
456    /// - Writes a range tombstone to the WAL.
457    /// - The tombstone shadows point entries with lower LSNs.
458    pub fn delete_range(&self, start: Vec<u8>, end: Vec<u8>) -> Result<(), MemtableError> {
459        trace!(
460            "delete_range() started, start key: {}, end key: {}",
461            HexKey(&start),
462            HexKey(&end)
463        );
464
465        if start.is_empty() || end.is_empty() {
466            return Err(MemtableError::Internal(
467                "Start or end key is empty".to_string(),
468            ));
469        }
470
471        let lsn = self.next_lsn.fetch_add(1, Ordering::SeqCst);
472        let timestamp = Self::current_timestamp();
473
474        let record_size = std::mem::size_of::<RangeTombstone>() + start.len() + end.len();
475        let record = Record::RangeDelete {
476            start: start.clone(),
477            end: end.clone(),
478            lsn,
479            timestamp,
480        };
481
482        let mut guard = self.inner.write().unwrap();
483
484        if guard.approximate_size + record_size > guard.write_buffer_size {
485            return Err(MemtableError::FlushRequired);
486        }
487
488        // 1. Wal first (crash safety)
489        self.wal.append(&record)?;
490
491        // 2. In-memory update
492        let value = RangeTombstone {
493            start: start.to_vec(),
494            end: end.to_vec(),
495            lsn,
496            timestamp,
497        };
498
499        guard
500            .range_tombstones
501            .entry(start.to_vec())
502            .or_default()
503            .insert(Reverse(value.lsn), value);
504
505        guard.approximate_size += record_size;
506
507        trace!(
508            "Delete operation completed with LSN: {}, start key: {}, end key: {}",
509            lsn,
510            HexKey(&start),
511            HexKey(&end),
512        );
513
514        Ok(())
515    }

517    /// Retrieves the latest visible value for a key.
518    ///
519    /// Resolution rules:
520    /// 1. Select highest-LSN point entry
521    /// 2. Check all covering range tombstones
522    /// 3. If a tombstone has a higher LSN, the key is considered deleted
523    ///
524    /// # Returns
525    /// - `Ok(Some(value))` if visible
526    /// - `Ok(None)` if deleted or not present
527    pub fn get(&self, key: &[u8]) -> Result<MemtableGetResult, MemtableError> {
528        trace!("get() started, key: {}", HexKey(key));
529
530        let guard = self.inner.read().map_err(|_| {
531            error!("Read-write lock poisoned during get");
532            MemtableError::Internal("RwLock poisoned".into())
533        })?;
534
535        // Check if key exists as a point entry
536        let point_opt = guard
537            .tree
538            .get(key)
539            .and_then(|versions| versions.values().next());
540
541        // Check if key matches any range tombstones.
542        // For each start key, we check ALL versions (not just the highest-LSN)
543        // because a narrower tombstone with a higher LSN might not cover the
544        // queried key while a wider tombstone with a lower LSN does.
545        let mut covering_tombstone_lsn: Option<u64> = None;
546        for (_start, versions) in guard.range_tombstones.range(..=key.to_vec()) {
547            for tombstone in versions.values() {
548                if tombstone.start.as_slice() <= key && key < tombstone.end.as_slice() {
549                    covering_tombstone_lsn = Some(
550                        covering_tombstone_lsn
551                            .map(|lsn| lsn.max(tombstone.lsn))
552                            .unwrap_or(tombstone.lsn),
553                    );
554                    // Found the highest-LSN covering tombstone for this start
555                    // key — no need to check lower-LSN versions for the same
556                    // start key (they can only have equal or lower LSN).
557                    break;
558                }
559            }
560        }
561
562        match (point_opt, covering_tombstone_lsn) {
563            // No point entry and no tombstone → key not found
564            (None, None) => Ok(MemtableGetResult::NotFound),
565
566            // No point entry but covered by range tombstone
567            (None, Some(_)) => Ok(MemtableGetResult::RangeDelete),
568
569            // Point entry exists, no covering tombstone
570            (Some(point), None) => {
571                if point.is_delete {
572                    Ok(MemtableGetResult::Delete)
573                } else {
574                    Ok(MemtableGetResult::Put(
575                        point
576                            .value
577                            .clone()
578                            .expect("Non-delete point entry must have a value"),
579                    ))
580                }
581            }
582
583            // Both point entry and tombstone exist → compare LSNs
584            (Some(point), Some(tombstone_lsn)) => {
585                if tombstone_lsn > point.lsn {
586                    Ok(MemtableGetResult::RangeDelete)
587                } else if point.is_delete {
588                    Ok(MemtableGetResult::Delete)
589                } else {
590                    Ok(MemtableGetResult::Put(
591                        point
592                            .value
593                            .clone()
594                            .expect("Non-delete point entry must have a value"),
595                    ))
596                }
597            }
598        }
599    }

601    /// Performs an ordered range scan over `[start, end)`.
602    ///
603    /// Each key is resolved against:
604    /// - its latest point entry
605    /// - all applicable range tombstones
606    ///
607    /// Deleted keys are omitted from the result.
608    ///
609    /// # Complexity
610    /// O(N * R) where R is the number of overlapping range tombstones.
611    pub fn scan(
612        &self,
613        start: &[u8],
614        end: &[u8],
615    ) -> Result<impl Iterator<Item = Record>, MemtableError> {
616        trace!(
617            "scan() started with range. Start key: {} end key: {}",
618            HexKey(start),
619            HexKey(end)
620        );
621
622        if start >= end {
623            return Ok(Vec::new().into_iter());
624        }
625
626        let guard = self.inner.read().map_err(|_| {
627            error!("Read-write lock poisoned during scan");
628            MemtableError::Internal("RwLock poisoned".into())
629        })?;
630
631        let mut out = Vec::new();
632
633        // 1) Collect point entries
634        for (key, versions) in guard.tree.range(start.to_vec()..end.to_vec()) {
635            for entry in versions.values() {
636                let record = if entry.is_delete {
637                    Record::Delete {
638                        key: key.clone(),
639                        lsn: entry.lsn,
640                        timestamp: entry.timestamp,
641                    }
642                } else {
643                    Record::Put {
644                        key: key.clone(),
645                        value: entry.value.clone().unwrap(),
646                        lsn: entry.lsn,
647                        timestamp: entry.timestamp,
648                    }
649                };
650
651                out.push(record);
652            }
653        }
654
655        // 2) Collect range tombstones
656        for (_start_key, versions) in guard.range_tombstones.iter() {
657            for tombstone in versions.values() {
658                // Check if tombstone overlaps scan range
659                if tombstone.end.as_slice() <= start || tombstone.start.as_slice() >= end {
660                    continue;
661                }
662
663                let record = Record::RangeDelete {
664                    start: tombstone.start.clone(),
665                    end: tombstone.end.clone(),
666                    lsn: tombstone.lsn,
667                    timestamp: tombstone.timestamp,
668                };
669
670                out.push(record);
671            }
672        }
673
674        // 3) Sort stream: key ASC, lsn DESC
675        out.sort_by(|a, b| {
676            let ka = a.key();
677            let kb = b.key();
678
679            match ka.cmp(kb) {
680                std::cmp::Ordering::Equal => b.lsn().cmp(&a.lsn()), // Descending LSN
681                other => other,
682            }
683        });
684
685        Ok(out.into_iter())
686    }

688    /// Returns a logical snapshot of the memtable suitable for flushing.
689    ///
690    /// The iterator emits:
691    /// - The latest version of every point key (put or delete)
692    /// - **All** range tombstones
693    ///
694    /// # Guarantees
695    /// - No filtering based on tombstone interaction
696    /// - Returned records are sufficient to rebuild the same state
697    /// - Does not mutate in-memory state
698    ///
699    /// # Intended Use
700    /// This iterator is consumed by the SSTable writer.
701    pub fn iter_for_flush(&self) -> Result<impl Iterator<Item = Record>, MemtableError> {
702        let guard = self.inner.read().map_err(|_| {
703            error!("Read-write lock poisoned during iter_for_flush");
704            MemtableError::Internal("Read-write lock poisoned".into())
705        })?;
706
707        let mut records = Vec::new();
708
709        for (key, versions) in guard.tree.iter() {
710            if let Some(entry) = versions.values().next() {
711                let record = if entry.is_delete {
712                    Record::Delete {
713                        key: key.clone(),
714                        lsn: entry.lsn,
715                        timestamp: entry.timestamp,
716                    }
717                } else {
718                    Record::Put {
719                        key: key.clone(),
720                        value: entry.value.clone().unwrap(),
721                        lsn: entry.lsn,
722                        timestamp: entry.timestamp,
723                    }
724                };
725                records.push(record);
726            }
727        }
728
729        for (start, versions) in guard.range_tombstones.iter() {
730            for entry in versions.values() {
731                let record = Record::RangeDelete {
732                    start: start.clone(),
733                    end: entry.end.clone(),
734                    lsn: entry.lsn,
735                    timestamp: entry.timestamp,
736                };
737                records.push(record);
738            }
739        }
740
741        Ok(records.into_iter())
742    }

754    /// Override the current LSN counter with a recovered value.
755    ///
756    /// # Safety / Rules
757    /// - Must only be called during recovery **before any writes**.
758    /// - Ensures that future LSNs always increase beyond recovered state.
759    pub fn inject_max_lsn(&self, lsn: u64) {
760        // next_lsn always points to the *next available* LSN
761        self.next_lsn.store(lsn.saturating_add(1), Ordering::SeqCst);
762    }

764    /// Returns the highest assigned LSN so far.
765    ///
766    /// This returns `next_lsn - 1`, since `next_lsn` always stores the next unused sequence number.
767    pub fn max_lsn(&self) -> u64 {
768        self.next_lsn.load(Ordering::SeqCst).saturating_sub(1)
769    }
770
771    /// Returns the current system timestamp in nanoseconds.
772    ///
773    /// Used to tag entries for ordering and diagnostics.
774    fn current_timestamp() -> u64 {
775        SystemTime::now()
776            .duration_since(UNIX_EPOCH)
777            .expect("system clock before UNIX epoch")
778            .as_nanos() as u64
779    }
780}

782// ------------------------------------------------------------------------------------------------
783// Frozen Memtable
784// ------------------------------------------------------------------------------------------------
785
786/// An immutable, read-only view of a memtable.
787///
788/// A frozen memtable:
789/// - Exposes only read APIs
790/// - Retains ownership of the WAL
791/// - Prevents further mutation by construction
792///
793/// This type represents a memtable that is in the process of being flushed
794/// to an on-disk SSTable.
795pub struct FrozenMemtable {
796    pub memtable: Memtable,
797    pub creation_timestamp: u64,
798}
799
800impl FrozenMemtable {
801    /// Creates a new frozen memtable by opening and replaying a WAL.
802    pub fn new(memtable: Memtable) -> Self {
803        Self {
804            memtable,
805            creation_timestamp: Memtable::current_timestamp(),
806        }
807    }
808
809    /// Retrieves the latest visible value for a key.
810    pub fn get(&self, key: &[u8]) -> Result<MemtableGetResult, MemtableError> {
811        self.memtable.get(key)
812    }
813
814    /// Performs a range scan over the frozen memtable.
815    pub fn scan(
816        &self,
817        start: &[u8],
818        end: &[u8],
819    ) -> Result<impl Iterator<Item = Record>, MemtableError> {
820        self.memtable.scan(start, end)
821    }
822
823    /// Returns all records required to materialize this memtable into an SSTable.
824    pub fn iter_for_flush(&self) -> Result<impl Iterator<Item = Record>, MemtableError> {
825        self.memtable.iter_for_flush()
826    }
827
828    /// Returns the highest assigned LSN so far.
829    pub fn max_lsn(&self) -> u64 {
830        self.memtable.max_lsn()
831    }
832}

834// ------------------------------------------------------------------------------------------------
835// Tracing Helper
836// ------------------------------------------------------------------------------------------------
837
838struct HexKey<'a>(&'a [u8]);
839
840impl<'a> std::fmt::Display for HexKey<'a> {
841    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
842        if self.0.len() <= 32 {
843            for byte in self.0 {
844                write!(f, "{:02x}", byte)?;
845            }
846        } else {
847            for byte in &self.0[..16] {
848                write!(f, "{:02x}", byte)?;
849            }
850            write!(f, "...[{} bytes]", self.0.len())?;
851        }
852        Ok(())
853    }
854}
