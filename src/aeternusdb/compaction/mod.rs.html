1//! # Compaction Module
2//!
3//! Implements three compaction strategies for the LSM storage engine:
4//!
5//! ## Minor Compaction (Size-Tiered)
6//!
7//! Groups SSTables into **size buckets** and merges similarly-sized tables
8//! when a bucket exceeds `min_threshold` entries. Deduplicates point entries
9//! (keeps highest LSN per key) but **preserves all tombstones** — both point
10//! and range — because other SSTables outside the merge set may still hold
11//! covered data.
12//!
13//! ## Tombstone Compaction (Per-SSTable GC)
14//!
15//! Rewrites a single SSTable to remove **point tombstones** that are provably
16//! unnecessary. Uses bloom filters on other SSTables to determine safety:
17//!
18//! - If no other SSTable's bloom says "maybe" for the key → drop the tombstone.
19//! - If bloom says "maybe" and `tombstone_bloom_fallback = true` → do actual `get()`
20//!   to resolve false positives.
21//! - If `tombstone_range_drop = true` → scan older SSTables to check
22//!   whether a range tombstone still covers any live keys.
23//!
24//! ## Major Compaction (Full Merge)
25//!
26//! User-triggered via `Engine::major_compact()`. Merges **all** SSTables into
27//! one, applying range tombstones actively to suppress covered point entries.
28//! All spent tombstones (both point and range) are dropped from the output
29//! since the entire SSTable set is merged — no data can resurrect.
30//!
31//! ## Code organization
32//!
33//! The module separates strategy-specific logic (bucketing, selection) from
34//! shared execution primitives (merge, dedup, build). This allows future
35//! strategies (e.g., leveled compaction) to reuse the merge/build plumbing.
36

50// ------------------------------------------------------------------------------------------------
51// CompactionStrategy trait
52// ------------------------------------------------------------------------------------------------
53
54/// A uniform interface for compaction strategies.
55///
56/// Each strategy receives the current set of SSTables (by reference), a
57/// mutable manifest for atomic metadata updates, the data directory, and
58/// the engine configuration. It returns:
59///
60/// - `Ok(Some(result))` — compaction was performed; the caller should
61///   update in-memory state using [`CompactionResult`].
62/// - `Ok(None)` — nothing to compact (thresholds not met, etc.).
63pub trait CompactionStrategy {
64    /// Execute one round of compaction, if the strategy's preconditions
65    /// are met. Implementations must be idempotent — calling when there
66    /// is nothing to do should simply return `Ok(None)`.
67    fn compact(
68        &self,
69        sstables: &[Arc<SSTable>],
70        manifest: &mut Manifest,
71        data_dir: &str,
72        config: &EngineConfig,
73    ) -> Result<Option<CompactionResult>, CompactionError>;
74}
75
76// ------------------------------------------------------------------------------------------------
77// CompactionStrategyType — config-level strategy selector
78// ------------------------------------------------------------------------------------------------
79
80/// Selects which compaction strategy family the engine should use.
81///
82/// Stored in [`DbConfig`](crate::DbConfig) and used by the engine to
83/// obtain the concrete strategy implementations for minor, tombstone,
84/// and major compaction.
85///
86/// # Example
87///
88/// ```rust
89/// use aeternusdb::{DbConfig, CompactionStrategyType};
90///
91/// let config = DbConfig {
92///     compaction_strategy: CompactionStrategyType::Stcs,
93///     ..DbConfig::default()
94/// };
95/// ```
96#[derive(Debug, Clone, Copy, PartialEq, Eq)]
97pub enum CompactionStrategyType {
98    /// Size-Tiered Compaction Strategy (STCS).
99    ///
100    /// Groups SSTables into size buckets and merges similarly-sized tables.
101    /// Good for write-heavy workloads with moderate space amplification.
102    Stcs,
103}
104
105impl CompactionStrategyType {
106    /// Returns the minor compaction strategy for this family.
107    pub fn minor(&self) -> Box<dyn CompactionStrategy> {
108        match self {
109            Self::Stcs => Box::new(stcs::MinorCompaction),
110        }
111    }
112
113    /// Returns the tombstone compaction strategy for this family.
114    pub fn tombstone(&self) -> Box<dyn CompactionStrategy> {
115        match self {
116            Self::Stcs => Box::new(stcs::TombstoneCompaction),
117        }
118    }
119
120    /// Returns the major compaction strategy for this family.
121    pub fn major(&self) -> Box<dyn CompactionStrategy> {
122        match self {
123            Self::Stcs => Box::new(stcs::MajorCompaction),
124        }
125    }
126}

132/// Result of a compaction execution — enough information to update the
133/// manifest and in-memory SSTable list.
134pub struct CompactionResult {
135    /// SSTable IDs that were consumed (to be removed from manifest).
136    pub removed_ids: Vec<u64>,
137
138    /// Path of the newly built SSTable (if any). `None` when all entries
139    /// were eliminated (e.g., all tombstones dropped in major compaction).
140    pub new_sst_path: Option<String>,
141
142    /// The ID allocated for the new SSTable (if one was produced).
143    pub new_sst_id: Option<u64>,
144}

146// ------------------------------------------------------------------------------------------------
147// Dedup logic — shared between minor and tombstone compaction
148// ------------------------------------------------------------------------------------------------
149
150/// Deduplicates a merge iterator stream into separate point entries
151/// and range tombstones.
152///
153/// For each unique key, keeps only the version with the highest LSN.
154/// **All tombstones (point and range) are preserved** — this is safe
155/// for minor compaction where other SSTables may hold covered data.
156pub fn dedup_records(
157    merge_iter: impl Iterator<Item = Record>,
158) -> (Vec<PointEntry>, Vec<RangeTombstone>) {
159    let mut point_entries = Vec::new();
160    let mut range_tombstones = Vec::new();
161    let mut last_key: Option<Vec<u8>> = None;
162
163    for record in merge_iter {
164        match record {
165            Record::RangeDelete {
166                start,
167                end,
168                lsn,
169                timestamp,
170            } => {
171                range_tombstones.push(RangeTombstone {
172                    start,
173                    end,
174                    lsn,
175                    timestamp,
176                });
177            }
178            Record::Put {
179                key,
180                value,
181                lsn,
182                timestamp,
183            } => {
184                if last_key.as_ref() == Some(&key) {
185                    continue; // Older version — skip
186                }
187                last_key = Some(key.clone());
188                point_entries.push(PointEntry {
189                    key,
190                    value: Some(value),
191                    lsn,
192                    timestamp,
193                });
194            }
195            Record::Delete {
196                key,
197                lsn,
198                timestamp,
199            } => {
200                if last_key.as_ref() == Some(&key) {
201                    continue; // Older version — skip
202                }
203                last_key = Some(key.clone());
204                point_entries.push(PointEntry {
205                    key,
206                    value: None,
207                    lsn,
208                    timestamp,
209                });
210            }
211        }
212    }
213
214    (point_entries, range_tombstones)
215}

221/// Creates scan iterators for the given SSTables covering their full key range.
222///
223/// Computes the min/max key bounds across all selected SSTables and returns
224/// a vec of boxed iterators that can be fed into [`MergeIterator`].
225///
226/// The returned iterators borrow the SSTables; iteration is **streaming**
227/// (block-by-block via mmap) so only one data block per SSTable is
228/// resident in memory at a time.
229pub fn full_range_scan_iters<'a>(
230    sstables: &'a [&'a SSTable],
231) -> Result<Vec<Box<dyn Iterator<Item = Record> + 'a>>, SSTableError> {
232    if sstables.is_empty() {
233        return Ok(Vec::new());
234    }
235
236    // Compute scan bounds from properties.
237    let min_key = sstables
238        .iter()
239        .map(|s| &s.properties.min_key)
240        .min()
241        .ok_or_else(|| SSTableError::Internal("empty sstables in full_range_scan".into()))?
242        .clone();
243
244    let mut max_key = sstables
245        .iter()
246        .map(|s| &s.properties.max_key)
247        .max()
248        .ok_or_else(|| SSTableError::Internal("empty sstables in full_range_scan".into()))?
249        .clone();
250    // Extend max_key past the actual max key to make it exclusive.
251    max_key.push(0xFF);
252
253    let mut iters: Vec<Box<dyn Iterator<Item = Record> + 'a>> = Vec::new();
254    for sst in sstables {
255        let scan = sst.scan(&min_key, &max_key)?;
256        iters.push(Box::new(scan));
257    }
258
259    Ok(iters)
260}

279// ------------------------------------------------------------------------------------------------
280// Finalize — shared build + manifest + cleanup
281// ------------------------------------------------------------------------------------------------
282
283/// Builds a new SSTable from the given entries, atomically updates the
284/// manifest, and deletes old SSTable files.
285///
286/// If both `point_entries` and `range_tombstones` are empty, no new SSTable
287/// is produced — old SSTables are simply removed.
288///
289/// This is the common tail shared by minor, tombstone, and major compaction.
290pub(crate) fn finalize_compaction(
291    manifest: &mut Manifest,
292    data_dir: &str,
293    removed_ids: Vec<u64>,
294    point_entries: Vec<PointEntry>,
295    range_tombstones: Vec<RangeTombstone>,
296) -> Result<CompactionResult, CompactionError> {
297    use std::fs;
298    use std::path::PathBuf;
299
300    if point_entries.is_empty() && range_tombstones.is_empty() {
301        // Nothing survived — just remove old SSTables from manifest.
302        info!(
303            removed_count = removed_ids.len(),
304            ?removed_ids,
305            "finalize: all entries eliminated, removing old SSTables"
306        );
307        manifest.apply_compaction(Vec::new(), removed_ids.clone())?;
308        manifest.checkpoint()?;
309
310        for id in &removed_ids {
311            let path = format!("{}/{}/{:06}.sst", data_dir, SSTABLE_DIR, id);
312            if let Err(e) = fs::remove_file(&path) {
313                tracing::warn!(id, %e, "failed to remove old SSTable file during compaction");
314            }
315        }
316
317        return Ok(CompactionResult {
318            removed_ids,
319            new_sst_path: None,
320            new_sst_id: None,
321        });
322    }
323
324    // Build new SSTable.
325    let new_sst_id = manifest.allocate_sst_id()?;
326    let new_sst_path = format!("{}/{}/{:06}.sst", data_dir, SSTABLE_DIR, new_sst_id);
327
328    let point_count = point_entries.len();
329    let range_count = range_tombstones.len();
330
331    debug!(
332        new_sst_id,
333        point_count,
334        range_count,
335        removed_count = removed_ids.len(),
336        path = %new_sst_path,
337        "finalize: building new SSTable"
338    );
339
340    sstable::SstWriter::new(&new_sst_path).build(
341        point_entries.into_iter(),
342        point_count,
343        range_tombstones.into_iter(),
344        range_count,
345    )?;
346
347    // Atomic manifest update: add new, remove old.
348    let new_entry = ManifestSstEntry {
349        id: new_sst_id,
350        path: PathBuf::from(&new_sst_path),
351    };
352    manifest.apply_compaction(vec![new_entry], removed_ids.clone())?;
353    manifest.checkpoint()?;
354
355    // Delete old SSTable files.
356    for id in &removed_ids {
357        let path = format!("{}/{}/{:06}.sst", data_dir, SSTABLE_DIR, id);
358        if let Err(e) = fs::remove_file(&path) {
359            tracing::warn!(id, %e, "failed to remove old SSTable file during compaction");
360        }
361    }
362
363    Ok(CompactionResult {
364        removed_ids,
365        new_sst_path: Some(new_sst_path),
366        new_sst_id: Some(new_sst_id),
367    })
