1//! Write-Ahead Logging (WAL) Module
2//!
3//! This module implements a **durable**, **append-only**, and **generic** Write-Ahead Log (WAL)
4//! suitable for embedded databases and key-value storage engines.  
5//! It provides **type-safe**, **CRC-protected**, and **thread-safe** persistence of arbitrary records
6//! that implement the [`WalData`] trait.
7//!
8//! ## Design Overview
9//!
10//! The WAL ensures crash recovery and corruption detection for any serializable record type
11//! (`MemTableRecord`, `ManifestRecord`, etc.). It uses [`crate::encoding`] for compact serialization
12//! and [`crc32fast`] for data integrity.
13//!
14//! Each record is appended sequentially to disk with atomic file syncs to ensure durability.
15//! The file handle is shared safely between threads using `Arc<Mutex<File>>`.
16//!
17//! # On-disk layout
18//!
19//! ```text
20//! [HEADER_BYTES][HEADER_CRC32_LE]
21//! [REC_LEN_LE][REC_BYTES][REC_CRC32_LE]
22//! [REC_LEN_LE][REC_BYTES][REC_CRC32_LE]
23//! ...
24//! ```
25//!
26//! - **Header** — a [`WalHeader`] structure followed by a 4-byte CRC32 checksum.
27//! - **Record** — consists of:
28//!   - 4-byte little-endian length prefix
29//!   - serialized record bytes (custom encoding format)
30//!   - 4-byte CRC32 checksum computed over `len || record_bytes`
31//!
32//! # Concurrency model
33//!
34//! - WAL access is **synchronized** via `Arc<Mutex<File>>`, ensuring consistent reads and writes.
35//! - Multiple components can safely share the same WAL — e.g. background compaction, recovery, and
36//!   write threads.
37//! - [`WalIter`] tracks its own logical offset, seeking before each read to avoid race conditions
38//!   with concurrent appenders.
39//!
40//! # Guarantees
41//!
42//! - **Durability:** Every `append()` is followed by an `fsync()` via [`File::sync_all`].  
43//! - **Integrity:** Both header and record checksums are verified during replay.  
44//! - **Corruption detection:** Replay stops at first failed checksum or truncated write.  
45//! - **Safety:** Thread-safe, generic over any [`crate::encoding`] `Encode`/`Decode` type.
46

77/// Errors returned by WAL operations.
78#[derive(Debug, Error)]
79#[non_exhaustive]
80pub enum WalError {
81    /// Underlying I/O error.
82    #[error("I/O error: {0}")]
83    Io(#[from] io::Error),
84
85    /// Encoding / decoding error.
86    #[error("Encoding error: {0}")]
87    Encoding(#[from] EncodingError),
88
89    /// Data integrity failure — checksum did not match.
90    #[error("Checksum mismatch")]
91    ChecksumMismatch,
92
93    /// Record exceeds the configured maximum size.
94    #[error("Record size exceeds limit ({0} bytes)")]
95    RecordTooLarge(usize),
96
97    /// Unexpected end-of-file during read.
98    #[error("Unexpected end of file")]
99    UnexpectedEof,
100
101    /// WAL header failed integrity validation.
102    #[error("Internal header: {0}")]
103    InvalidHeader(String),
104
105    /// Internal consistency or locking error.
106    #[error("Internal error: {0}")]
107    Internal(String),
108}

114/// Metadata written at the start of the WAL file.
115///
116/// This section validates the WAL’s identity and constraints.
117/// It is followed by a CRC32 checksum to protect against corruption.
118#[derive(Debug)]
119pub struct WalHeader {
120    /// Magic constant to identify WAL files (`b"AWAL"`).
121    magic: [u8; 4],
122
123    /// WAL format version.
124    version: u32,
125
126    /// Maximum record size (in bytes).
127    max_record_size: u32,
128
129    /// Monotonically-increasing WAL sequence number (segment id).
130    wal_seq: u64,
131}
132
133impl WalHeader {
134    /// Expected 4-byte magic constant.
135    pub const MAGIC: [u8; 4] = *b"AWAL";
136
137    /// Current supported version number.
138    pub const VERSION: u32 = 1;
139
140    /// Default maximum record size (1 MiB).
141    pub const DEFAULT_MAX_RECORD_SIZE: u32 = 1024 * 1024;
142
143    /// Creates a new [`WalHeader`] instance.
144    ///
145    /// # Parameters
146    /// - `max_record_size`: Maximum record size limit.
147    /// - `wal_seq`: WAL segment sequence number.
148    pub fn new(max_record_size: u32, wal_seq: u64) -> Self {
149        Self {
150            magic: Self::MAGIC,
151            version: Self::VERSION,
152            max_record_size,
153            wal_seq,
154        }
155    }
156
157    /// Encoded size of the header in bytes (without the trailing CRC).
158    ///
159    /// Layout: `magic(4) + version(4) + max_record_size(4) + wal_seq(8)` = 20.
160    pub const ENCODED_SIZE: usize = 4 + 4 + 4 + 8;
161
162    /// Total on-disk size of the header *including* its trailing CRC32.
163    pub const HEADER_DISK_SIZE: usize = Self::ENCODED_SIZE + U32_SIZE;
164
165    /// Returns the WAL segment sequence number.
166    #[allow(dead_code)]
167    pub fn wal_seq(&self) -> u64 {
168        self.wal_seq
169    }
170
171    /// Returns the maximum record size (in bytes).
172    #[allow(dead_code)]
173    pub fn max_record_size(&self) -> u32 {
174        self.max_record_size
175    }
176
177    /// Returns the WAL format version.
178    #[allow(dead_code)]
179    pub fn version(&self) -> u32 {
180        self.version
181    }
182}

184impl encoding::Encode for WalHeader {
185    fn encode_to(&self, buf: &mut Vec<u8>) -> Result<(), EncodingError> {
186        encoding::Encode::encode_to(&self.magic, buf)?;
187        encoding::Encode::encode_to(&self.version, buf)?;
188        encoding::Encode::encode_to(&self.max_record_size, buf)?;
189        encoding::Encode::encode_to(&self.wal_seq, buf)?;
190        Ok(())
191    }
192}
193
194impl encoding::Decode for WalHeader {
195    fn decode_from(buf: &[u8]) -> Result<(Self, usize), EncodingError> {
196        let mut offset = 0;
197        let (magic, n) = <[u8; 4]>::decode_from(&buf[offset..])?;
198        offset += n;
199        let (version, n) = u32::decode_from(&buf[offset..])?;
200        offset += n;
201        let (max_record_size, n) = u32::decode_from(&buf[offset..])?;
202        offset += n;
203        let (wal_seq, n) = u64::decode_from(&buf[offset..])?;
204        offset += n;
205        Ok((
206            Self {
207                magic,
208                version,
209                max_record_size,
210                wal_seq,
211            },
212            offset,
213        ))
214    }
215}

237/// A generic, thread-safe Write-Ahead Log for durable record storage.
238///
239/// See the [module-level documentation](self) for more details on format,
240/// concurrency, and guarantees.
241///
242/// # Type Parameters
243///
244/// * `T` — Any record type implementing [`WalData`].
245#[derive(Debug)]
246pub struct Wal<T: WalData> {
247    /// Thread-safe file handle for WAL operations.
248    inner_file: Arc<Mutex<File>>,
249
250    /// Path to the WAL file on disk.
251    path: PathBuf,
252
253    /// Persistent header with metadata and integrity info.
254    header: WalHeader,
255
256    /// Marker field to associate this WAL with the generic record type `T`.
257    _phantom: std::marker::PhantomData<T>,
258}
259
260impl<T: WalData> Wal<T> {
261    /// Open or create a WAL file at the given path.
262    ///
263    /// # Parameters
264    /// - `path`: Path to the WAL file.
265    /// - `max_record_size`: Optional custom maximum record size.
266    ///
267    /// # Returns
268    /// A [`Wal`] instance, ready for appending or replaying records.
269    pub fn open<P: AsRef<Path>>(path: P, max_record_size: Option<u32>) -> Result<Self, WalError> {
270        let path_ref = path.as_ref();
271        let mut file = OpenOptions::new()
272            .create(true)
273            .read(true)
274            .append(true)
275            .open(path_ref)?;
276
277        let wal_seq = Self::parse_seq_from_path(path_ref)
278            .ok_or(WalError::Internal("WAL name incorrect".into()))?;
279
280        // If file is empty, create and write a new header.
281        let header = if file.metadata()?.len() == 0 {
282            let header = WalHeader::new(
283                max_record_size.unwrap_or(WalHeader::DEFAULT_MAX_RECORD_SIZE),
284                wal_seq,
285            );
286
287            write_header(&mut file, &header)?;
288            file.sync_all()?;
289
290            info!(path = %path_ref.display(), seq = wal_seq, "WAL created with new header");
291
292            header
293        } else {
294            // Existing WAL → read and validate header + checksum.
295            file.seek(SeekFrom::Start(0))?;
296
297            let header = read_and_validate_header(&mut file)?;
298
299            if header.wal_seq != wal_seq {
300                return Err(WalError::InvalidHeader("sequence number mismatch".into()));
301            }
302
303            debug!(
304                path = %path_ref.display(),
305                max_record_size = header.max_record_size,
306                seq = header.wal_seq,
307                "WAL header validated"
308            );
309
310            header
311        };
312
313        info!(path = %path_ref.display(), seq = header.wal_seq, "WAL opened");
314
315        Ok(Self {
316            inner_file: Arc::new(Mutex::new(file)),
317            path: path_ref.to_path_buf(),
318            header,
319            _phantom: std::marker::PhantomData,
320        })
321    }
322
323    /// Parse `wal_seq` from filename if it matches `<seq>.log`.
324    fn parse_seq_from_path(path: &Path) -> Option<u64> {
325        let name = path.file_name().and_then(OsStr::to_str)?;
326        // Expect pattern 000001.log or 1.log etc.
327        let seq_str = name.strip_suffix(".log")?;
328        seq_str.parse::<u64>().ok()
329    }
330
331    /// Appends a single record to the WAL.
332    ///
333    /// The record is serialized using [`crate::encoding`] and written as:
334    /// `[u32 len LE][record_bytes][u32 crc32 LE]`,
335    /// where the CRC is computed over the `len || record_bytes`.
336    ///
337    /// # Parameters
338    /// - `record`: Reference to the record implementing [`WalData`].
339    pub fn append(&self, record: &T) -> Result<(), WalError> {
340        let record_bytes = encoding::encode_to_vec(record)?;
341        let record_len = u32::try_from(record_bytes.len())
342            .map_err(|_| WalError::RecordTooLarge(record_bytes.len()))?;
343
344        if record_len > self.header.max_record_size {
345            return Err(WalError::RecordTooLarge(record_len as usize));
346        }
347
348        let len_bytes = record_len.to_le_bytes();
349        let checksum = compute_crc(&[&len_bytes, &record_bytes]);
350
351        // Lock and append atomically (from user's perspective).
352        let mut guard = self
353            .inner_file
354            .lock()
355            .map_err(|_| WalError::Internal("Mutex poisoned".into()))?;
356
357        guard.write_all(&len_bytes)?;
358        guard.write_all(&record_bytes)?;
359        guard.write_all(&checksum.to_le_bytes())?;
360        guard.sync_all()?;
361
362        trace!(
363            len = record_len,
364            crc = format_args!("{checksum:08x}"),
365            "WAL record appended"
366        );
367        Ok(())
368    }
369
370    /// Returns an iterator that replays all valid records from the WAL.
371    ///
372    /// The iterator reads the WAL sequentially, verifies CRC checksums,
373    /// and decodes each entry into its original record type `T`.
374    pub fn replay_iter(&self) -> Result<WalIter<T>, WalError> {
375        debug!(path = %self.path.display(), "WAL replay started");
376
377        let start_offset = WalHeader::HEADER_DISK_SIZE as u64;
378
379        Ok(WalIter {
380            file: Arc::clone(&self.inner_file),
381            offset: start_offset,
382            max_record_size: self.header.max_record_size as usize,
383            _phantom: std::marker::PhantomData,
384        })
385    }
386
387    /// Truncate (clear) the WAL and rewrite header.
388    ///
389    /// After truncation, WAL contains only the header and its checksum.
390    pub fn truncate(&mut self) -> Result<(), WalError> {
391        let mut guard = self
392            .inner_file
393            .lock()
394            .map_err(|_| WalError::Internal("Mutex poisoned".into()))?;
395
396        guard.set_len(0)?;
397        guard.seek(SeekFrom::Start(0))?;
398
399        write_header(&mut *guard, &self.header)?;
400        guard.sync_all()?;
401
402        info!(path = %self.path.display(), "WAL truncated");
403        Ok(())
404    }
405
406    /// Rotates to a new WAL segment with the next sequence number.
407    ///
408    /// Syncs the current WAL, opens a new WAL file with `wal_seq + 1`,
409    /// and replaces `self` with the new instance.
410    ///
411    /// Returns the new WAL sequence number.
412    #[allow(dead_code)]
413    pub fn rotate_next(&mut self) -> Result<u64, WalError> {
414        {
415            let guard = self
416                .inner_file
417                .lock()
418                .map_err(|_| WalError::Internal("Mutex poisoned".into()))?;
419            guard.sync_all()?;
420        }
421
422        let next_seq = self
423            .header
424            .wal_seq
425            .checked_add(1)
426            .ok_or_else(|| WalError::Internal("WAL sequence number overflow".into()))?;
427
428        let cur_path = PathBuf::from(&self.path);
429        let dir = cur_path.parent().unwrap_or_else(|| Path::new("."));
430        let next_path = dir.join(format!("{next_seq:06}.log"));
431
432        let new_wal = Wal::<T>::open(&next_path, Some(self.header.max_record_size))?;
433        *self = new_wal;
434
435        Ok(next_seq)
436    }
437
438    /// Get the path of the underlying WAL file.
439    #[allow(dead_code)]
440    pub fn path(&self) -> &Path {
441        &self.path
442    }
443
444    /// Returns the WAL segment sequence number.
445    pub fn wal_seq(&self) -> u64 {
446        self.header.wal_seq
447    }
448
449    /// Returns the configured maximum record size.
450    #[allow(dead_code)]
451    pub fn max_record_size(&self) -> u32 {
452        self.header.max_record_size
453    }
454
455    /// Returns the current on-disk file size in bytes.
456    #[allow(dead_code)]
457    pub fn file_size(&self) -> Result<u64, WalError> {
458        let guard = self
459            .inner_file
460            .lock()
461            .map_err(|_| WalError::Internal("Mutex poisoned".into()))?;
462        Ok(guard.metadata()?.len())
463    }
464}

466impl<T: WalData> Drop for Wal<T> {
467    fn drop(&mut self) {
468        match self.inner_file.lock() {
469            Ok(guard) => {
470                if let Err(e) = guard.sync_all() {
471                    error!(path = %self.path.display(), error = %e, "WAL sync failed on drop");
472                }
473            }
474            Err(poisoned) => {
475                let file = poisoned.into_inner();
476                if let Err(e) = file.sync_all() {
477                    error!(path = %self.path.display(), error = %e, "WAL sync failed on drop (poisoned lock)");
478                } else {
479                    warn!(path = %self.path.display(), "WAL recovered and synced after poisoned lock");
480                }
481            }
482        }
483    }
484}

490/// Streaming WAL replay iterator.
491///
492/// `WalIter` reads records sequentially from the WAL file and yields decoded `T` values.
493/// It is designed to:
494///
495/// - **Stream** records without allocating the entire WAL into memory (one record at a time).
496/// - **Share** the WAL file safely with appenders by holding an `Arc<Mutex<File>>`.
497/// - **Detect corruption** and truncated writes using CRC32 checksums and length bounds.
498///
499/// # Lifetime & ownership
500///
501/// The iterator holds an `Arc` reference to the underlying file handle. This means
502/// it can **outlive** the [`Wal`] that created it — the file will remain open until
503/// all iterators (and the WAL itself) are dropped.
504pub struct WalIter<T: WalData> {
505    /// Shared file handle protected by a mutex.
506    file: Arc<Mutex<File>>,
507
508    /// Current byte offset within WAL file.
509    offset: u64,
510
511    /// Maximum allowed record size.
512    max_record_size: usize,
513
514    /// Marker field to associate this WAL iterator with the generic record type `T`.
515    _phantom: std::marker::PhantomData<T>,
516}
517
518impl<T: WalData> std::fmt::Debug for WalIter<T> {
519    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
520        f.debug_struct("WalIter")
521            .field("offset", &self.offset)
522            .field("max_record_size", &self.max_record_size)
523            .finish_non_exhaustive()
524    }
525}
526
527impl<T: WalData> Iterator for WalIter<T> {
528    type Item = Result<T, WalError>;
529
530    fn next(&mut self) -> Option<Self::Item> {
531        // Lock only during the read of one record to reduce contention.
532        let mut guard = match self.file.lock() {
533            Ok(g) => g,
534            Err(_) => return Some(Err(WalError::Internal("Mutex poisoned".into()))),
535        };
536
537        // Seek to our logical offset for deterministic reads.
538        if let Err(e) = guard.seek(SeekFrom::Start(self.offset)) {
539            return Some(Err(WalError::Io(e)));
540        }
541
542        // Read length prefix (4 bytes).
543        let mut len_bytes = [0u8; U32_SIZE];
544
545        match guard.read_exact(&mut len_bytes) {
546            Ok(_) => {}
547            Err(e) if e.kind() == io::ErrorKind::UnexpectedEof => {
548                trace!(offset = self.offset, "WAL replay reached end of file");
549                return None;
550            }
551            Err(e) => return Some(Err(WalError::Io(e))),
552        }
553
554        let record_len = u32::from_le_bytes(len_bytes) as usize;
555        if record_len > self.max_record_size {
556            return Some(Err(WalError::RecordTooLarge(record_len)));
557        }
558
559        trace!(offset = self.offset, len = record_len, "WAL reading record");
560
561        // Read record bytes.
562        let mut record_bytes = vec![0u8; record_len];
563        if let Err(e) = guard.read_exact(&mut record_bytes) {
564            if e.kind() == io::ErrorKind::UnexpectedEof {
565                warn!(
566                    offset = self.offset,
567                    len = record_len,
568                    "WAL truncated record (partial payload)"
569                );
570                return Some(Err(WalError::UnexpectedEof));
571            }
572            return Some(Err(WalError::Io(e)));
573        }
574
575        // Read stored checksum.
576        let mut checksum_bytes = [0u8; U32_SIZE];
577        if let Err(e) = guard.read_exact(&mut checksum_bytes) {
578            if e.kind() == io::ErrorKind::UnexpectedEof {
579                warn!(
580                    offset = self.offset,
581                    len = record_len,
582                    "WAL truncated record (partial checksum)"
583                );
584                return Some(Err(WalError::UnexpectedEof));
585            }
586            return Some(Err(WalError::Io(e)));
587        }
588        let stored_checksum = u32::from_le_bytes(checksum_bytes);
589
590        // Update offset for next iteration using current file cursor position.
591        if let Ok(pos) = guard.stream_position() {
592            self.offset = pos;
593        }
594
595        // Verify checksum over [len || record_bytes].
596        if let Err(e) = verify_crc(&[&len_bytes, &record_bytes], stored_checksum) {
597            warn!(
598                offset = self.offset,
599                len = record_len,
600                "WAL record checksum mismatch"
601            );
602            return Some(Err(e));
603        }
604
605        // Decode the record payload.
606        match encoding::decode_from_slice::<T>(&record_bytes) {
607            Ok((record, _)) => Some(Ok(record)),
608            Err(e) => Some(Err(WalError::Encoding(e))),
609        }
610    }
611}

617/// Writes a [`WalHeader`] followed by its CRC32 checksum, then syncs.
618fn write_header<W: Write>(writer: &mut W, header: &WalHeader) -> Result<(), WalError> {
619    let header_bytes = encoding::encode_to_vec(header)?;
620    let checksum = compute_crc(&[&header_bytes]);
621
622    writer.write_all(&header_bytes)?;
623    writer.write_all(&checksum.to_le_bytes())?;
624
625    // Sync if the writer is a File (trait objects won't have sync_all, but
626    // our callers always follow up with their own sync when needed).
627    Ok(())
628}
629
630/// Reads and validates a [`WalHeader`] from the current file position.
631///
632/// Checks CRC, magic, and version. Does **not** validate `wal_seq` (the
633/// caller must do that, since the expected sequence depends on context).
634fn read_and_validate_header<R: Read>(reader: &mut R) -> Result<WalHeader, WalError> {
635    let mut header_bytes = vec![0u8; WalHeader::ENCODED_SIZE];
636    reader.read_exact(&mut header_bytes)?;
637
638    let mut checksum_bytes = [0u8; U32_SIZE];
639    reader.read_exact(&mut checksum_bytes)?;
640    let stored_checksum = u32::from_le_bytes(checksum_bytes);
641
642    verify_crc(&[&header_bytes], stored_checksum)
643        .map_err(|_| WalError::InvalidHeader("header checksum mismatch".into()))?;
644
645    let (header, _) = encoding::decode_from_slice::<WalHeader>(&header_bytes)?;
646
647    if header.magic != WalHeader::MAGIC {
648        return Err(WalError::InvalidHeader("bad magic".into()));
649    }
650    if header.version != WalHeader::VERSION {
651        return Err(WalError::InvalidHeader(format!(
652            "unsupported version {}",
653            header.version
654        )));
655    }
656
657    Ok(header)
658}

664/// Computes a CRC32 checksum over one or more byte slices.
665fn compute_crc(parts: &[&[u8]]) -> u32 {
666    let mut hasher = Crc32::new();
667    for part in parts {
668        hasher.update(part);
669    }
670    hasher.finalize()
671}
672
673/// Verifies that the CRC32 over the given byte slices matches `expected`.
674fn verify_crc(parts: &[&[u8]], expected: u32) -> Result<(), WalError> {
675    let computed = compute_crc(parts);
676    if computed != expected {
677        return Err(WalError::ChecksumMismatch);
678    }
679    Ok(())
