1//! Write-Ahead Logging (WAL) Module
2//!
3//! This module implements a **durable**, **append-only**, and **generic** Write-Ahead Log (WAL)
4//! suitable for embedded databases and key-value storage engines.  
5//! It provides **type-safe**, **CRC-protected**, and **thread-safe** persistence of arbitrary records
6//! that implement the [`WalData`] trait.
7//!
8//! //! ## Design Overview
9//!
10//! The WAL ensures crash recovery and corruption detection for any serializable record type
11//! (`MemTableRecord`, `ManifestRecord`, etc.). It uses [`crate::encoding`] for compact serialization
12//! and [`crc32fast`] for data integrity.
13//!
14//! Each record is appended sequentially to disk with atomic file syncs to ensure durability.
15//! The file handle is shared safely between threads using `Arc<Mutex<File>>`.
16//!
17//! # On-disk layout
18//!
19//! ```text
20//! [HEADER_BYTES][HEADER_CRC32_LE]
21//! [REC_LEN_LE][REC_BYTES][REC_CRC32_LE]
22//! [REC_LEN_LE][REC_BYTES][REC_CRC32_LE]
23//! ...
24//! ```
25//!
26//! - **Header** — a [`WalHeader`] structure followed by a 4-byte CRC32 checksum.
27//! - **Record** — consists of:
28//!   - 4-byte little-endian length prefix
29//!   - serialized record bytes (custom encoding format)
30//!   - 4-byte CRC32 checksum computed over `len || record_bytes`
31//!
32//! # Concurrency model
33//!
34//! - WAL access is **synchronized** via `Arc<Mutex<File>>`, ensuring consistent reads and writes.
35//! - Multiple components can safely share the same WAL — e.g. background compaction, recovery, and
36//!   write threads.
37//! - [`WalIter`] tracks its own logical offset, seeking before each read to avoid race conditions
38//!   with concurrent appenders.
39//!
40//! # Guarantees
41//!
42//! - **Durability:** Every `append()` is followed by an `fsync()` via [`File::sync_all`].  
43//! - **Integrity:** Both header and record checksums are verified during replay.  
44//! - **Corruption detection:** Replay stops at first failed checksum or truncated write.  
45//! - **Safety:** Thread-safe, generic over any [`crate::encoding`] `Encode`/`Decode` type.
46

77/// Errors returned by WAL operations.
78#[derive(Debug, Error)]
79pub enum WalError {
80    /// Underlying I/O error.
81    #[error("I/O error: {0}")]
82    Io(#[from] io::Error),
83
84    /// Encoding / decoding error.
85    #[error("Encoding error: {0}")]
86    Encoding(#[from] EncodingError),
87
88    /// Data integrity failure — checksum did not match.
89    #[error("Checksum mismatch")]
90    ChecksumMismatch,
91
92    /// Record exceeds the configured maximum size.
93    #[error("Record size exceeds limit ({0} bytes)")]
94    RecordTooLarge(usize),
95
96    /// Unexpected end-of-file during read.
97    #[error("Unexpected end of file")]
98    UnexpectedEof,
99
100    /// WAL header failed integrity validation.
101    #[error("Internal header: {0}")]
102    InvalidHeader(String),
103
104    /// Internal consistency or locking error.
105    #[error("Internal error: {0}")]
106    Internal(String),
107}

113/// Metadata written at the start of the WAL file.
114///
115/// This section validates the WAL’s identity and constraints.
116/// It is followed by a CRC32 checksum to protect against corruption.
117#[derive(Debug)]
118pub struct WalHeader {
119    /// Magic constant to identify WAL files (`b"AWAL"`).
120    pub magic: [u8; 4],
121
122    /// WAL format version.
123    pub version: u32,
124
125    /// Maximum record size (in bytes).
126    pub max_record_size: u32,
127
128    /// Monotonically-increasing WAL sequence number (segment id).
129    pub wal_seq: u64,
130}
131
132impl WalHeader {
133    /// Expected 4-byte magic constant.
134    pub const MAGIC: [u8; 4] = *b"AWAL";
135
136    /// Current supported version number.
137    pub const VERSION: u32 = 1;
138
139    /// Default maximum record size (1 MiB).
140    pub const DEFAULT_MAX_RECORD_SIZE: u32 = 1024 * 1024;
141
142    /// Creates a new [`WalHeader`] instance.
143    ///
144    /// # Parameters
145    /// - `max_record_size`: Maximum record size limit.
146    pub fn new(max_record_size: u32, wal_seq: u64) -> Self {
147        Self {
148            magic: Self::MAGIC,
149            version: Self::VERSION,
150            max_record_size,
151            wal_seq,
152        }
153    }
154}
155
156impl encoding::Encode for WalHeader {
157    fn encode_to(&self, buf: &mut Vec<u8>) -> Result<(), EncodingError> {
158        encoding::Encode::encode_to(&self.magic, buf)?;
159        encoding::Encode::encode_to(&self.version, buf)?;
160        encoding::Encode::encode_to(&self.max_record_size, buf)?;
161        encoding::Encode::encode_to(&self.wal_seq, buf)?;
162        Ok(())
163    }
164}
165
166impl encoding::Decode for WalHeader {
167    fn decode_from(buf: &[u8]) -> Result<(Self, usize), EncodingError> {
168        let mut offset = 0;
169        let (magic, n) = <[u8; 4]>::decode_from(&buf[offset..])?;
170        offset += n;
171        let (version, n) = u32::decode_from(&buf[offset..])?;
172        offset += n;
173        let (max_record_size, n) = u32::decode_from(&buf[offset..])?;
174        offset += n;
175        let (wal_seq, n) = u64::decode_from(&buf[offset..])?;
176        offset += n;
177        Ok((
178            Self {
179                magic,
180                version,
181                max_record_size,
182                wal_seq,
183            },
184            offset,
185        ))
186    }
187}

209/// A generic, thread-safe Write-Ahead Log for durable record storage.
210///
211/// See the [module-level documentation](self) for more details on format,
212/// concurrency, and guarantees.
213///
214/// # Type Parameters
215///
216/// * `T` — Any record type implementing [`WalData`].
217#[derive(Debug)]
218pub struct Wal<T: WalData> {
219    /// Thread-safe file handle for WAL operations.
220    inner_file: Arc<Mutex<File>>,
221
222    /// Path to the WAL file on disk.
223    path: String,
224
225    /// Persistent header with metadata and integrity info.
226    pub header: WalHeader,
227
228    /// Marker field to associate this WAL with the generic record type `T`.
229    _phantom: std::marker::PhantomData<T>,
230}
231
232impl<T: WalData> Wal<T> {
233    /// Open or create a WAL file at the given path.
234    ///
235    /// # Parameters
236    /// - `path`: Path to the WAL file.
237    /// - `max_record_size`: Optional custom maximum record size.
238    ///
239    /// # Returns
240    /// A [`Wal`] instance, ready for appending or replaying records.
241    pub fn open<P: AsRef<Path>>(path: P, max_record_size: Option<u32>) -> Result<Self, WalError> {
242        let path_ref = path.as_ref();
243        let mut file = OpenOptions::new()
244            .create(true)
245            .read(true)
246            .append(true)
247            .open(path_ref)?;
248
249        let wal_seq = Self::parse_seq_from_path(path_ref)
250            .ok_or(WalError::Internal("WAL name incorrect".into()))?;
251
252        // If file is empty, create and write a new header.
253        let header = if file.metadata()?.len() == 0 {
254            let header = WalHeader::new(
255                max_record_size.unwrap_or(WalHeader::DEFAULT_MAX_RECORD_SIZE),
256                wal_seq,
257            );
258
259            let header_bytes = encoding::encode_to_vec(&header)?;
260
261            let mut hasher = Crc32::new();
262            hasher.update(&header_bytes);
263            let checksum = hasher.finalize();
264
265            file.write_all(&header_bytes)?;
266            file.write_all(&checksum.to_le_bytes())?;
267            file.sync_all()?;
268
269            info!("Created new WAL header at {}", path_ref.display());
270
271            header
272        } else {
273            // Existing WAL → read and validate header + checksum.
274            file.seek(SeekFrom::Start(0))?;
275
276            let sample = WalHeader::new(WalHeader::DEFAULT_MAX_RECORD_SIZE, 0);
277            let sample_bytes = encoding::encode_to_vec(&sample)?;
278            let header_len = sample_bytes.len();
279
280            let mut header_bytes = vec![0u8; header_len];
281            file.read_exact(&mut header_bytes)?;
282
283            let mut checksum_bytes = [0u8; U32_SIZE];
284            file.read_exact(&mut checksum_bytes)?;
285            let stored_checksum = u32::from_le_bytes(checksum_bytes);
286
287            let mut hasher = Crc32::new();
288            hasher.update(&header_bytes);
289            let computed_checksum = hasher.finalize();
290
291            if stored_checksum != computed_checksum {
292                return Err(WalError::InvalidHeader("Header checksum mismatched".into()));
293            }
294
295            let (header, _) = encoding::decode_from_slice::<WalHeader>(&header_bytes)?;
296
297            if header.magic != WalHeader::MAGIC {
298                return Err(WalError::InvalidHeader("Bad magic".into()));
299            }
300            if header.version != WalHeader::VERSION {
301                return Err(WalError::InvalidHeader(format!(
302                    "Unsupported version {}",
303                    header.version
304                )));
305            }
306
307            if header.wal_seq != wal_seq {
308                return Err(WalError::InvalidHeader("Invalid sequence number".into()));
309            }
310
311            info!(
312                "Loaded WAL header from {} (max_record_size={})",
313                path_ref.display(),
314                header.max_record_size
315            );
316
317            header
318        };
319
320        info!("Opened WAL file at {}", path_ref.display());
321
322        Ok(Self {
323            inner_file: Arc::new(Mutex::new(file)),
324            path: path_ref.display().to_string(),
325            header,
326            _phantom: std::marker::PhantomData,
327        })
328    }
329
330    /// Parse `wal_seq` from filename if it matches `wal-<seq>.log`.
331    fn parse_seq_from_path(path: &Path) -> Option<u64> {
332        let name = path.file_name().and_then(OsStr::to_str)?;
333        // Expect pattern wal-000001.log or wal-1.log etc.
334        if let Some(seq_str) = name
335            .strip_prefix("wal-")
336            .and_then(|s| s.strip_suffix(".log"))
337        {
338            return seq_str.parse::<u64>().ok();
339        }
340        None
341    }
342
343    /// Appends a single record to the WAL.
344    ///
345    /// The record is serialized using [`crate::encoding`] and written as:
346    /// `[u32 len LE][record_bytes][u32 crc32 LE]`,
347    /// where the CRC is computed over the `len || record_bytes`.
348    ///
349    /// # Parameters
350    /// - `record`: Reference to the record implementing [`WalData`].
351    pub fn append(&self, record: &T) -> Result<(), WalError> {
352        trace!("Appending record: {:?}", record,);
353
354        let record_bytes = encoding::encode_to_vec(record)?;
355        let record_len = u32::try_from(record_bytes.len())
356            .map_err(|_| WalError::RecordTooLarge(record_bytes.len()))?;
357
358        if record_len > self.header.max_record_size {
359            return Err(WalError::RecordTooLarge(record_len as usize));
360        }
361
362        // Compute checksum over [len_le || record_bytes]
363        let mut hasher = Crc32::new();
364        hasher.update(&record_len.to_le_bytes());
365        hasher.update(&record_bytes);
366        let checksum = hasher.finalize();
367
368        // Lock and append atomically (from user's perspective).
369        let mut guard = self
370            .inner_file
371            .lock()
372            .map_err(|_| WalError::Internal("Mutex poisoned".into()))?;
373
374        guard.write_all(&record_len.to_le_bytes())?;
375        guard.write_all(&record_bytes)?;
376        guard.write_all(&checksum.to_le_bytes())?;
377        guard.sync_all()?;
378
379        info!(
380            "Appended record of length {} with checksum {:08x}",
381            record_len, checksum
382        );
383        Ok(())
384    }
385
386    /// Returns an iterator that replays all valid records from the WAL.
387    ///
388    /// The iterator reads the WAL sequentially, verifies CRC checksums,
389    /// and decodes each entry into its original record type `T`.
390    pub fn replay_iter(&self) -> Result<WalIter<T>, WalError> {
391        info!("Starting WAL replay from file: {}", self.path);
392
393        let header_bytes = encoding::encode_to_vec(&self.header)?;
394        let start_offset = (header_bytes.len() + U32_SIZE) as u64;
395
396        Ok(WalIter {
397            file: Arc::clone(&self.inner_file),
398            offset: start_offset,
399            max_record_size: self.header.max_record_size as usize,
400            _phantom: std::marker::PhantomData,
401        })
402    }
403
404    /// Truncate (clear) the WAL and rewrite header.
405    ///
406    /// After truncation, WAL contains only the header and its checksum.
407    pub fn truncate(&mut self) -> Result<(), WalError> {
408        let mut guard = self
409            .inner_file
410            .lock()
411            .map_err(|_| WalError::Internal("Mutex poisoned".into()))?;
412
413        guard.set_len(0)?;
414        guard.seek(SeekFrom::Start(0))?;
415
416        let header_bytes = encoding::encode_to_vec(&self.header)?;
417
418        let mut hasher = Crc32::new();
419        hasher.update(&header_bytes);
420        let checksum = hasher.finalize();
421
422        guard.write_all(&header_bytes)?;
423        guard.write_all(&checksum.to_le_bytes())?;
424        guard.sync_all()?;
425
426        info!("Truncated WAL file: {}", self.path);
427        Ok(())
428    }
429
430    /// Rotates to a new WAL segment with the next sequence number.
431    ///
432    /// Syncs the current WAL, opens a new WAL file with `wal_seq + 1`,
433    /// and replaces `self` with the new instance.
434    ///
435    /// Returns the new WAL sequence number.
436    pub fn rotate_next(&mut self) -> Result<u64, WalError> {
437        {
438            let guard = self
439                .inner_file
440                .lock()
441                .map_err(|_| WalError::Internal("Mutex poisoned".into()))?;
442            guard.sync_all()?;
443        }
444
445        let next_seq = self.header.wal_seq.saturating_add(1);
446
447        let cur_path = PathBuf::from(&self.path);
448        let dir = cur_path.parent().unwrap_or_else(|| Path::new("."));
449        let next_path = dir.join(format!("wal-{next_seq:06}.log"));
450
451        let mut new_wal = Wal::<T>::open(&next_path, Some(self.header.max_record_size))?;
452        new_wal.header.wal_seq = next_seq;
453        *self = new_wal;
454
455        Ok(next_seq)
456    }
457
458    /// Get the path of the underlying WAL file.
459    pub fn path(&self) -> &str {
460        &self.path
461    }
462}

464impl<T: WalData> Drop for Wal<T> {
465    fn drop(&mut self) {
466        match self.inner_file.lock() {
467            Ok(guard) => {
468                if let Err(e) = guard.sync_all() {
469                    error!("Failed to sync WAL on drop: {}", e);
470                }
471            }
472            Err(poisoned) => {
473                let file = poisoned.into_inner();
474                if let Err(e) = file.sync_all() {
475                    error!("Failed to sync WAL (poisoned) on drop: {}", e);
476                } else {
477                    warn!("Recovered and synced WAL after poisoned lock");
478                }
479            }
480        }
481    }
482}

488/// Streaming WAL replay iterator.
489///
490/// `WalIter` reads records sequentially from the WAL file and yields decoded `T` values.
491/// It is designed to:
492///
493/// - **Stream** records without allocating the entire WAL into memory (one record at a time).
494/// - **Share** the WAL file safely with appenders by holding an `Arc<Mutex<File>>`.
495/// - **Detect corruption** and truncated writes using CRC32 checksums and length bounds.
496pub struct WalIter<T: WalData> {
497    /// Shared file handle protected by a mutex.
498    file: Arc<Mutex<File>>,
499
500    /// Current byte offset within WAL file.
501    offset: u64,
502
503    /// Maximum allowed record size.
504    max_record_size: usize,
505
506    /// Marker field to associate this WAL iterator with the generic record type `T`.
507    _phantom: std::marker::PhantomData<T>,
508}
509
510impl<T: WalData> Iterator for WalIter<T> {
511    type Item = Result<T, WalError>;
512
513    fn next(&mut self) -> Option<Self::Item> {
514        // Lock only during the read of one record to reduce contention.
515        let mut guard = match self.file.lock() {
516            Ok(g) => g,
517            Err(_) => return Some(Err(WalError::Internal("Mutex poisoned".into()))),
518        };
519
520        // Seek to our logical offset for deterministic reads.
521        if let Err(e) = guard.seek(SeekFrom::Start(self.offset)) {
522            return Some(Err(WalError::Io(e)));
523        }
524
525        // Read length prefix (4 bytes).
526        let mut len_bytes = [0u8; U32_SIZE];
527
528        match guard.read_exact(&mut len_bytes) {
529            Ok(_) => {}
530            Err(e) if e.kind() == io::ErrorKind::UnexpectedEof => {
531                trace!("End of WAL reached");
532                return None;
533            }
534            Err(e) => return Some(Err(WalError::Io(e))),
535        }
536
537        let record_len = u32::from_le_bytes(len_bytes) as usize;
538        if record_len > self.max_record_size {
539            return Some(Err(WalError::RecordTooLarge(record_len)));
540        }
541
542        trace!("Reading record of length {}", record_len);
543
544        // Read record bytes.
545        let mut record_bytes = vec![0u8; record_len];
546        if let Err(e) = guard.read_exact(&mut record_bytes) {
547            if e.kind() == io::ErrorKind::UnexpectedEof {
548                error!("Truncated WAL record detected");
549                return Some(Err(WalError::UnexpectedEof));
550            }
551            return Some(Err(WalError::Io(e)));
552        }
553
554        // Read stored checksum.
555        let mut checksum_bytes = [0u8; U32_SIZE];
556        if let Err(e) = guard.read_exact(&mut checksum_bytes) {
557            if e.kind() == io::ErrorKind::UnexpectedEof {
558                error!("Truncated WAL record detected");
559                return Some(Err(WalError::UnexpectedEof));
560            }
561            return Some(Err(WalError::Io(e)));
562        }
563        let stored_checksum = u32::from_le_bytes(checksum_bytes);
564
565        // Update offset for next iteration using current file cursor position.
566        if let Ok(pos) = guard.stream_position() {
567            self.offset = pos;
568        }
569
570        // Verify checksum over [len || record_bytes].
571        let mut hasher = Crc32::new();
572        hasher.update(&len_bytes);
573        hasher.update(&record_bytes);
574        let computed_checksum = hasher.finalize();
575
576        if stored_checksum != computed_checksum {
577            error!("Checksum mismatch for record of length {}", record_len);
578            return Some(Err(WalError::ChecksumMismatch));
579        }
580
581        // Decode the record payload.
582        match encoding::decode_from_slice::<T>(&record_bytes) {
583            Ok((record, _)) => Some(Ok(record)),
584            Err(e) => Some(Err(WalError::Encoding(e))),
585        }
586    }
